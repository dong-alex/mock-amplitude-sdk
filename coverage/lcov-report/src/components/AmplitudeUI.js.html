import React from 'react';
import { withAmplitudeContext } from "./AmplitudeContext";
 
/**
 * A simple UI that will wrap the amplitude provider and provide the necessary functionality to log events
 */
const AmplitudeUI = (props) => {
 
  const { apiKey, handleLogEvent, handleAPIKeyCreate } = props.context;
 
  const API_KEY = "ASK9302KJALILMNSOPHHSK";
 
  const handleSessionStart = () => {
    handleAPIKeyCreate(API_KEY);
  };
 
  const handleLogClick = () => {
    handleLogEvent('test log event');
  };
 
  return (
    <div data-test-id="amplitude-ui">
      <label>I am an example page</label>
      <button onClick={handleSessionStart}> Start Amplitude Session</button>
      {apiKey ? (
        <div data-test-id="message">There is an api key and it is {apiKey}</div>
      ) : (
        <div data-test-id="no-api-key-message">
          There is no api-key associated yet
        </div>
      )}
      <button onClick={handleLogClick}>Log Event</button>
    </div>
  );
};
 
/**
 * Calls withAmplitudeContext will wrap your component with the necessary context to use the SDK in your component
 */
 export default withAmplitudeContext(AmplitudeUI);
